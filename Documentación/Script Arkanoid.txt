CREATE TABLE PLAYER(
idPlayer SERIAL PRIMARY KEY,
nickname VARCHAR(15) UNIQUE NOT NULL
);

CREATE TABLE SCORE(
idScore SERIAL PRIMARY KEY,
idPlayer INT NOT NULL,
score INT NOT NULL,
CONSTRAINT fk_player_score
FOREIGN KEY (idPlayer) REFERENCES PLAYER(idPlayer) 
);
